CREATE OR REPLACE PROCEDURE UPDATEDATA(IID IN INT,QUA IN INT) IS
BEGIN
UPDATE SHOP SET QUANTITY=QUA WHERE ITEM_ID=IID;
IF SQL%NOTFOUND THEN
 DBMS_OUTPUT.PUT_LINE('ITEM NOT FOUND');
ELSE
  DBMS_OUTPUT.PUT_LINE('ITEM QUANTITY UPDATED');
END IF;
END;

DECLARE
ITEMID INT;
QTY INT;
BEGIN
ITEMID:=&ITEM_ID;
QTY:=&NEW_QUANTITY;
UPDATEDATA(ITEMID,QTY);
END;
