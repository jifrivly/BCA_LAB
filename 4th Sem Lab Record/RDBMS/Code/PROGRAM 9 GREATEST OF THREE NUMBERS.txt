RDBMS PROGRAM 9-GREATEST OF THREE NUMBERS

DECLARE
NUM1 INT;
NUM2 INT;
NUM3 INT;
BEGIN
NUM1:=&NUMBER1;
NUM2:=&NUMBER2;
NUM3:=&NUMBER3;
IF NUM1>NUM2 THEN
  IF NUM1>NUM3 THEN
    DBMS_OUTPUT.PUT_LINE(NUM1||' IS GREATEST');
  ELSE
    DBMS_OUTPUT.PUT_LINE(NUM3||' IS GREATEST');
  END IF;
ELSE
  IF NUM2>NUM3 THEN
    DBMS_OUTPUT.PUT_LINE(NUM2||' IS GREATEST');
  ELSE
  DBMS_OUTPUT.PUT_LINE(NUM3||' IS GREATES');
  END IF;
END IF;
END;
/
